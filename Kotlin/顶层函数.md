fun millis2Date(millis: Long): Date? {
    return Date(millis)
}

fun longToString(s: Long, type: String): String {
    var ret = ""
    var strTime: String? = s.toString()
    if (strTime != null) {
        if (strTime.length == 10) {
            strTime = (Integer.valueOf(strTime) * 1000L).toString()
        }
        val date = Date(java.lang.Long.valueOf(strTime))
        ret = format(date, type)
    }
    return ret
}

fun format(date: Date, type: String): String {
    val sdf = SimpleDateFormat(type)
    return sdf.format(date)
}


fun dip2px(dipValue: Float): Int {
    val scale = get().resources.displayMetrics.density
    return (dipValue * scale + 0.5f).toInt()
}

fun Int.dpToPx(displayMetrics: DisplayMetrics): Int = (this * displayMetrics.density).toInt()
